    <script>
        function login(event) {
            event.preventDefault();  

            const name = document.getElementById('name').value;
            const planet = document.getElementById('planet').value;

            if (!name || !planet) {
                document.getElementById('errorMessage').textContent = 'Kérjük, töltse ki mindkét mezőt!';
                return;
            }

            sessionStorage.setItem("user", JSON.stringify({ name, planet }));

            const outerSolarSystemPlanets = ['pluto', 'comet', 'haumea', 'eris', 'makemake', 'sedna'];
            const innerSolarSystemPlanets = ['sun', 'mercury', 'venus', 'earth', 'mars', 'jupiter', 'saturn', 'uranus', 'neptune'];

            let redirectPage;
            if (innerSolarSystemPlanets.includes(planet)) {
                redirectPage = 'index.html';  
            } else if (outerSolarSystemPlanets.includes(planet)) {
                redirectPage = 'index2.html';  
            } else {
                document.getElementById('errorMessage').textContent = 'Érvénytelen bolygó választás!';
                return;
            }

            window.location.href = redirectPage;
        }
    </script>
